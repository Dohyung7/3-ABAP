FUNCTION YF_GET_MATERIAL_INFO_206.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(I_MATNR) TYPE  MATNR
*"  EXPORTING
*"     REFERENCE(ES_MARA) TYPE  YS206_0020
*"----------------------------------------------------------------------

*       Global data declarations

CLEAR ES_MARA.

  SELECT SINGLE MATNR ERSDA MTART MATKL ERNAM MBRSH MEINS
       INTO CORRESPONDING FIELDS OF ES_MARA
       FROM MARA
      WHERE MATNR = I_MATNR.

  SELECT SINGLE MAKTX
    INTO ES_MARA-MAKTX
    FROM MAKT
   WHERE MATNR = I_MATNR
     AND SPRAS = SY-LANGU.



ENDFUNCTION.
